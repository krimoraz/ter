\begin{center}
  \textit{L'usage de la calculatrice est fortement recommandé.}
  \\
  \textit{La qualité globale de la copie sera appréciée suivant un
  système bonus/malus.}
\end{center}

\emph{Compétences :}
\begin{itemize}
  \item Étudier la limite d'une somme, d'un produit ou d'un quotient de
    deux suites ;
  \item Savoir mener un raisonnement par récurrence ;
  \item Construire un arbre pondéré de probabilités.
\end{itemize}

\begin{exercice}[Limite de suite]~

  \begin{enumerate}
    \item Donner la limite de la suite définie par $\forall n\in\N,\ u_n
      = n^2 - 3n$.
    \item Donner la limite de la suite définie par $\forall n\in\N,\ v_n
      = \dfrac{\sin(n^2 + 1)}{\sqrt{n+1}}$.
  \end{enumerate}
\end{exercice}

\begin{exercice}[Récurrence]~

  Démontrer par récurrence que $\forall n \in \N,\ 1 + 2 + 3 + \cdots +
  n = \dfrac{n(n+1)}2$
\end{exercice}

\begin{exercice}[Probabilités conditionnelles]~

  À la suite d'un sondage effectué à propos de la construction d'un
  barrage, on estime que :
  \begin{itemize}
    \item 65\% de la population concernée est contre la construction, et
      parmi ces opposants, 70\% sont des écologistes.
    \item parmi les personnes qui ne sont pas opposés à la construction,
      20\% sont des écologistes.
  \end{itemize}
  On interroge une personne au hasard.

  \begin{enumerate}
    \item Calculer la probabilité qu'elle soit opposée à la construction
      du barrage et écologiste.
    \item Calculer la probabilité qu'elle ne soit pas opposé à la
      construction du barrage et soit écologiste.
    \item Déduisez-en la probabilité qu'une personne interrogée soit
      écologiste.
  \end{enumerate}
\end{exercice}
