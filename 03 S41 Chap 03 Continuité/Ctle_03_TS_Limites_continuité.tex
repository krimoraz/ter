  \begin{multicols}{3}
    \begin{enumerate}
      \item $\lim_{x\to -\infty} x^3 + 3x^2 + 2x +1$
      \item $\lim_{x\to+\infty} \frac{x^2 - 3x + 4}{2x^2 - 3}$
      \item $\lim_{\substack{x\to 5\\x \leqslant 5}} \frac{\sin(x) -
        \cos(x)}{x^2 - 25}$
    \end{enumerate}
  \end{multicols}

\end{exercice}
